        // Инициализация Telegram Web App
        const tg = window.Telegram.WebApp;
        tg.expand(); // Разворачиваем на весь экран Айфона

        // Подтягиваем имя из телеграма
        if (tg.initDataUnsafe && tg.initDataUnsafe.user) {
            document.getElementById('username').innerText = "@" + tg.initDataUnsafe.user.username;
        } else {
            document.getElementById('username').innerText = "@subaru_owner";
        }

        // Логика игры (пока локальная, без сервера)
        let coins = 500;
        let rust = 80;
        let oil = 40;

        function updateUI() {
            document.getElementById('coins-val').innerText = coins;
            document.getElementById('rust-text').innerText = rust + "%";
            document.getElementById('rust-bar').style.width = rust + "%";
            document.getElementById('oil-text').innerText = oil + "%";
            document.getElementById('oil-bar').style.width = oil + "%";
        }

        function cleanRust() {
            if (coins >= 50 && rust > 0) {
                coins -= 50;
                rust = Math.max(0, rust - 20);
                updateUI();
                confetti({ wallet: true, particleCount: 40, spread: 60, origin: { y: 0.8 } });
            } else if (rust === 0) {
                tg.showAlert("Кузов уже блестит!");
            } else {
                tg.showAlert("Не хватает Subaru Coins!");
            }
        }

        function changeOil() {
            if (coins >= 100 && oil < 100) {
                coins -= 100;
                oil = Math.min(100, oil + 30);
                updateUI();
                tg.HapticFeedback.notificationOccurred('success'); // Вибрация на Айфоне!
            } else if (oil === 100) {
                tg.showAlert("Масло уже свежее!");
            } else {
                tg.showAlert("Не хватает Subaru Coins!");
            }
        }
